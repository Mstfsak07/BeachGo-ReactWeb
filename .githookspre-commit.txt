$review = git diff --cached | claude @"
[audit + redteam]

Bu değişiklikleri incele.
- kritik bug var mı
- null reference riski var mı
- güvenlik sorunu var mı

Ciddi sorun varsa BLOCK yaz.
Sorun yoksa OK yaz.
"@

$review | Out-File .claude-review.txt

if ($review -match "BLOCK") {
    Write-Host $review
    exit 1
}
