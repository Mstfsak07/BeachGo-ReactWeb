### `/BeachGo/api/Services/OtpService.cs`
```csharp
using BeachGo.Api.Data;
using BeachGo.Api.Models;
using Microsoft.EntityFrameworkCore;

namespace BeachGo.Api.Services;

public class OtpService : IOtpService
{
    private readonly ApplicationDbContext _context;
    private readonly ILogger<OtpService> _logger;

    public OtpService(ApplicationDbContext context, ILogger<OtpService> logger)
    {
        _context = context;
        _logger = logger;
    }

    public async Task<string> GenerateAndSaveCodeAsync(string userId, string type)
    {
        // Mevcut aktif kodlar횆짹 ge횄짠ersiz k횆짹l
        await InvalidateCodesAsync(userId, type);

        var code = Guid.NewGuid().ToString("N"); // 32 karakterlik token

        var verificationCode = new VerificationCode
        {
            UserId = userId,
            Code = code,
            Type = type,
            ExpiresAt = DateTime.UtcNow.AddHours(24),
            IsUsed = false,
            CreatedAt = DateTime.UtcNow
        };

        _context.VerificationCodes.Add(verificationCode);
        await _context.SaveChangesAsync();

        _logger.LogInformation("OTP generated for userId={UserId}, type={Type}", userId, type);
        return code;
    }

    public async Task<bool> ValidateCodeAsync(string userId, string code, string type)
    {
        var verificationCode = await _context.VerificationCodes
            .Where(v => v.UserId == userId
                     && v.Code == code
                     && v.Type == type
                     && !v.IsUsed
                     && v.ExpiresAt > DateTime.UtcNow)
            .FirstOrDefaultAsync();

        if (verificationCode == null)
        {
            _logger.LogWarning("Invalid or expired OTP for userId={UserId}, type={Type}", userId, type);
            return false;
        }

        verificationCode.IsUsed = true;
        await _context.SaveChangesAsync();

        return true;
    }

    public async Task InvalidateCodesAsync(string userId, string type)
    {
        var codes = await _context.VerificationCodes
            .Where(v => v.UserId == userId && v.Type == type && !v.IsUsed)
            .ToListAsync();

        foreach (var c in codes)
            c.IsUsed = true;

        if (codes.Any())
            await _context.SaveChangesAsync();
    }
}

```csharp
using BeachGo.Api.Models;
using Microsoft.AspNetCore.Identity;

namespace BeachGo.Api.Services;

public class AuthService : IAuthService
{
    private readonly UserManager<ApplicationUser> _userManager;
    private readonly IOtpService _otpService;
    private readonly IEmailService _emailService;
    private readonly IJwtService _jwtService;
    private readonly ILogger<AuthService> _logger;

    public AuthService(
        UserManager<ApplicationUser> userManager,
        IOtpService otpService,
        IEmailService emailService,
        IJwtService jwtService,
        ILogger<AuthService> logger)
    {
        _userManager = userManager;
        _otpService = otpService;
        _emailService = emailService;
        _jwtService = jwtService;
        _logger = logger;
    }

    public async Task<AuthResponse> RegisterAsync(RegisterRequest request)
    {
        var existingUser = await _userManager.FindByEmailAsync(request.Email);
        if (existingUser != null)
            return new AuthResponse { Success = false, Message = "Bu email adresi zaten kullan횆짹mda." };

        var user = new ApplicationUser
        {
            UserName = request.Email,
            Email = request.Email,
            FullName = request.FullName,
            PhoneNumber = request.PhoneNumber,
            EmailConfirmed = false
        };

        var result = await _userManager.CreateAsync(user, request.Password);
        if (!result.Succeeded)
            return new AuthResponse
            {
                Success = false,
                Message = string.Join(", ", result.Errors.Select(e => e.Description))
            };

        var token = await _otpService.GenerateAndSaveCodeAsync(user.Id, "email_verification");
        await _emailService.SendVerificationEmailAsync(user.Email!, user.Id, token);

        return new AuthResponse
        {
            Success = true,
            Message = "Kay횆짹t ba횇혺ar횆짹l횆짹. Email adresinizi do횆혺rulay횆짹n.",
            UserId = user.Id
        };
    }

    public async Task<AuthResponse> LoginAsync(LoginRequest request)
    {
        var user = await _userManager.FindByEmailAsync(request.Email);
        if (user == null)
            return new AuthResponse { Success = false, Message = "Email veya 횇혺ifre hatal횆짹." };

        var isPasswordValid = await _userManager.CheckPasswordAsync(user, request.Password);
        if (!isPasswordValid)
            return new AuthResponse { Success = false, Message = "Email veya 횇혺ifre hatal횆짹." };

        if (!user.EmailConfirmed)
            return new AuthResponse { Success = false, Message = "Email adresinizi do횆혺rulaman횆짹z gerekmektedir." };

        var token = _jwtService.GenerateToken(user);

        return new AuthResponse
        {
            Success = true,
            Token = token,
            UserId = user.Id,
            Email = user.Email
        };
    }

    public async Task<AuthResponse> VerifyEmailAsync(VerifyEmailRequest request)
    {
        var user = await _userManager.FindByIdAsync(request.UserId);
        if (user == null)
            return new AuthResponse { Success = false, Message = "Kullan횆짹c횆짹 bulunamad횆짹." };

        if (user.EmailConfirmed)
            return new AuthResponse { Success = true, Message = "Email zaten do횆혺rulanm횆짹횇혺." };

        var isValid = await _otpService.ValidateCodeAsync(request.UserId, request.Token, "email_verification");
        if (!isValid)
            return new AuthResponse { Success = false, Message = "Ge횄짠ersiz veya s횄쩌resi dolmu횇혺 do횆혺rulama kodu." };

        user.EmailConfirmed = true;
        await _userManager.UpdateAsync(user);

        return new AuthResponse { Success = true, Message = "Email ba횇혺ar횆짹yla do횆혺ruland횆짹." };
    }

    public async Task<AuthResponse> ForgotPasswordAsync(ForgotPasswordRequest request)
    {
        var user = await _userManager.FindByEmailAsync(request.Email);
        // G횄쩌venlik: kullan횆짹c횆짹 bulunamasa bile ba횇혺ar횆짹l횆짹 d횄쨋n
        if (user == null)
            return new AuthResponse { Success = true, Message = "횇혷ifre s횆짹f횆짹rlama linki g횄쨋nderildi." };

        var token = await _otpService.GenerateAndSaveCodeAsync(user.Id, "password_reset");
        await _emailService.SendPasswordResetEmailAsync(user.Email!, user.Id, token);

        return new AuthResponse { Success = true, Message = "횇혷ifre s횆짹f횆짹rlama linki email adresinize g횄쨋nderildi." };
    }

    public async Task<AuthResponse> ResetPasswordAsync(ResetPasswordRequest request)
    {
        var user = await _userManager.FindByEmailAsync(request.Email);
        if (user == null)
            return new AuthResponse { Success = false, Message = "Kullan횆짹c횆짹 bulunamad횆짹." };

        var isValid = await _otpService.ValidateCodeAsync(user.Id, request.Token, "password_reset");
        if (!isValid)
            return new AuthResponse { Success = false, Message = "Ge횄짠ersiz veya s횄쩌resi dolmu횇혺 token." };

        var removeResult = await _userManager.RemovePasswordAsync(user);
        if (!removeResult.Succeeded)
            return new AuthResponse { Success = false, Message = "횇혷ifre s횆짹f횆짹rlanamad횆짹." };

        var addResult = await _userManager.AddPasswordAsync(user, request.NewPassword);
        if (!addResult.Succeeded)
            return new AuthResponse
            {
                Success = false,
                Message = string.Join(", ", addResult.Errors.Select(e => e.Description))
            };

        return new AuthResponse { Success = true, Message = "횇혷ifreniz ba횇혺ar횆짹yla s횆짹f횆짹rland횆짹." };
    }

    public async Task<AuthResponse> ResendVerificationAsync(ResendVerificationRequest request)
    {
        var user = await _userManager.FindByEmailAsync(request.Email);
        if (user == null)
            return new AuthResponse { Success = true, Message = "Do횆혺rulama emaili g횄쨋nderildi." };

        if (user.EmailConfirmed)
            return new AuthResponse { Success = false, Message = "Email zaten do횆혺rulanm횆짹횇혺." };

        var token = await _otpService.GenerateAndSaveCodeAsync(user.Id, "email_verification");
        await _emailService.SendVerificationEmailAsync(user.Email!, user.Id, token);

        return new AuthResponse { Success = true, Message = "Do횆혺rulama emaili g횄쨋nderildi." };
    }
}

### `/BeachGo/api/Controllers/AuthController.cs`
```csharp
using BeachGo.Api.Models;
using BeachGo.Api.Services;
using FluentValidation;
using Microsoft.AspNetCore.Mvc;

namespace BeachGo.Api.Controllers;

[ApiController]
[Route("api/auth")]
public class AuthController : ControllerBase
{
    private readonly IAuthService _authService;
    private readonly IValidator<ForgotPasswordRequest> _forgotPasswordValidator;
    private readonly IValidator<ResetPasswordRequest> _resetPasswordValidator;
    private readonly IValidator<VerifyEmailRequest> _verifyEmailValidator;
    private readonly IValidator<ResendVerificationRequest> _resendVerificationValidator;

    public AuthController(
        IAuthService authService,
        IValidator<ForgotPasswordRequest> forgotPasswordValidator,
        IValidator<ResetPasswordRequest> resetPasswordValidator,
        IValidator<VerifyEmailRequest> verifyEmailValidator,
        IValidator<ResendVerificationRequest> resendVerificationValidator)
    {
        _authService = authService;
        _forgotPasswordValidator = forgotPasswordValidator;
        _resetPasswordValidator = resetPasswordValidator;
        _verifyEmailValidator = verifyEmailValidator;
        _resendVerificationValidator = resendVerificationValidator;
    }

    [HttpPost("register")]
    public async Task<IActionResult> Register([FromBody] RegisterRequest request)
    {
        var result = await _authService.RegisterAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);
    }

    [HttpPost("login")]
    public async Task<IActionResult> Login([FromBody] LoginRequest request)
    {
        var result = await _authService.LoginAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);
    }

    [HttpPost("verify-email")]
    public async Task<IActionResult> VerifyEmail([FromBody] VerifyEmailRequest request)
    {
        var validation = await _verifyEmailValidator.ValidateAsync(request);
        if (!validation.IsValid)
            return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

        var result = await _authService.VerifyEmailAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);
    }

    [HttpPost("forgot-password")]
    public async Task<IActionResult> ForgotPassword([FromBody] ForgotPasswordRequest request)
    {
        var validation = await _forgotPasswordValidator.ValidateAsync(request);
        if (!validation.IsValid)
            return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

        var result = await _authService.ForgotPasswordAsync(request);
        return Ok(result); // Her zaman 200 d횄쨋n (g횄쩌venlik)
    }

    [HttpPost("reset-password")]
    public async Task<IActionResult> ResetPassword([FromBody] ResetPasswordRequest request)
    {
        var validation = await _resetPasswordValidator.ValidateAsync(request);
        if (!validation.IsValid)
            return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

        var result = await _authService.ResetPasswordAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);
    }

    [HttpPost("resend-verification")]
    public async Task<IActionResult> ResendVerification([FromBody] ResendVerificationRequest request)
    {
        var validation = await _resendVerificationValidator.ValidateAsync(request);
        if (!validation.IsValid)
            return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

        var result = await _authService.ResendVerificationAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);
    }
}

```csharp
using BeachGo.Api.Data;
using BeachGo.Api.Models;
using Microsoft.EntityFrameworkCore;

namespace BeachGo.Api.Services;

public class OtpService : IOtpService
{
    private readonly ApplicationDbContext _context;
    private readonly ILogger<OtpService> _logger;

    public OtpService(ApplicationDbContext context, ILogger<OtpService> logger)
    {
        _context = context;
        _logger = logger;
    }

    public async Task<string> GenerateAndSaveCodeAsync(string userId, string type)
    {
        // Mevcut aktif kodlar횆짹 ge횄짠ersiz k횆짹l
        await InvalidateCodesAsync(userId, type);

        var code = Guid.NewGuid().ToString("N"); // 32 karakterlik token

        var verificationCode = new VerificationCode
        {
            UserId = userId,
            Code = code,
            Type = type,
            ExpiresAt = DateTime.UtcNow.AddHours(24),
            IsUsed = false,
            CreatedAt = DateTime.UtcNow
        };

        _context.VerificationCodes.Add(verificationCode);
        await _context.SaveChangesAsync();

        _logger.LogInformation("OTP generated for userId={UserId}, type={Type}", userId, type);
        return code;
    }

    public async Task<bool> ValidateCodeAsync(string userId, string code, string type)
    {
        var verificationCode = await _context.VerificationCodes
            .Where(v => v.UserId == userId
                     && v.Code == code
                     && v.Type == type
                     && !v.IsUsed
                     && v.ExpiresAt > DateTime.UtcNow)
            .FirstOrDefaultAsync();

        if (verificationCode == null)
        {
            _logger.LogWarning("Invalid or expired OTP for userId={UserId}, type={Type}", userId, type);
            return false;
        }

        verificationCode.IsUsed = true;
        await _context.SaveChangesAsync();

        return true;
    }

    public async Task InvalidateCodesAsync(string userId, string type)
    {
        var codes = await _context.VerificationCodes
            .Where(v => v.UserId == userId && v.Type == type && !v.IsUsed)
            .ToListAsync();

        foreach (var c in codes)
            c.IsUsed = true;

        if (codes.Any())
            await _context.SaveChangesAsync();
    }
}

```csharp
using BeachGo.Api.Models;
using Microsoft.AspNetCore.Identity;

namespace BeachGo.Api.Services;

public class AuthService : IAuthService
{
    private readonly UserManager<ApplicationUser> _userManager;
    private readonly IOtpService _otpService;
    private readonly IEmailService _emailService;
    private readonly IJwtService _jwtService;
    private readonly ILogger<AuthService> _logger;

    public AuthService(
        UserManager<ApplicationUser> userManager,
        IOtpService otpService,
        IEmailService emailService,
        IJwtService jwtService,
        ILogger<AuthService> logger)
    {
        _userManager = userManager;
        _otpService = otpService;
        _emailService = emailService;
        _jwtService = jwtService;
        _logger = logger;
    }

    public async Task<AuthResponse> RegisterAsync(RegisterRequest request)
    {
        var existingUser = await _userManager.FindByEmailAsync(request.Email);
        if (existingUser != null)
            return new AuthResponse { Success = false, Message = "Bu email adresi zaten kullan횆짹mda." };

        var user = new ApplicationUser
        {
            UserName = request.Email,
            Email = request.Email,
            FullName = request.FullName,
            PhoneNumber = request.PhoneNumber,
            EmailConfirmed = false
        };

        var result = await _userManager.CreateAsync(user, request.Password);
        if (!result.Succeeded)
            return new AuthResponse
            {
                Success = false,
                Message = string.Join(", ", result.Errors.Select(e => e.Description))
            };

        var token = await _otpService.GenerateAndSaveCodeAsync(user.Id, "email_verification");
        await _emailService.SendVerificationEmailAsync(user.Email!, user.Id, token);

        return new AuthResponse
        {
            Success = true,
            Message = "Kay횆짹t ba횇혺ar횆짹l횆짹. Email adresinizi do횆혺rulay횆짹n.",
            UserId = user.Id
        };
    }

    public async Task<AuthResponse> LoginAsync(LoginRequest request)
    {
        var user = await _userManager.FindByEmailAsync(request.Email);
        if (user == null)
            return new AuthResponse { Success = false, Message = "Email veya 횇혺ifre hatal횆짹." };

        var isPasswordValid = await _userManager.CheckPasswordAsync(user, request.Password);
        if (!isPasswordValid)
            return new AuthResponse { Success = false, Message = "Email veya 횇혺ifre hatal횆짹." };

        if (!user.EmailConfirmed)
            return new AuthResponse { Success = false, Message = "Email adresinizi do횆혺rulaman횆짹z gerekmektedir." };

        var token = _jwtService.GenerateToken(user);

        return new AuthResponse
        {
            Success = true,
            Token = token,
            UserId = user.Id,
            Email = user.Email
        };
    }

    public async Task<AuthResponse> VerifyEmailAsync(VerifyEmailRequest request)
    {
        var user = await _userManager.FindByIdAsync(request.UserId);
        if (user == null)
            return new AuthResponse { Success = false, Message = "Kullan횆짹c횆짹 bulunamad횆짹." };

        if (user.EmailConfirmed)
            return new AuthResponse { Success = true, Message = "Email zaten do횆혺rulanm횆짹횇혺." };

        var isValid = await _otpService.ValidateCodeAsync(request.UserId, request.Token, "email_verification");
        if (!isValid)
            return new AuthResponse { Success = false, Message = "Ge횄짠ersiz veya s횄쩌resi dolmu횇혺 do횆혺rulama kodu." };

        user.EmailConfirmed = true;
        await _userManager.UpdateAsync(user);

        return new AuthResponse { Success = true, Message = "Email ba횇혺ar횆짹yla do횆혺ruland횆짹." };
    }

    public async Task<AuthResponse> ForgotPasswordAsync(ForgotPasswordRequest request)
    {
        var user = await _userManager.FindByEmailAsync(request.Email);
        // G횄쩌venlik: kullan횆짹c횆짹 bulunamasa bile ba횇혺ar횆짹l횆짹 d횄쨋n
        if (user == null)
            return new AuthResponse { Success = true, Message = "횇혷ifre s횆짹f횆짹rlama linki g횄쨋nderildi." };

        var token = await _otpService.GenerateAndSaveCodeAsync(user.Id, "password_reset");
        await _emailService.SendPasswordResetEmailAsync(user.Email!, user.Id, token);

        return new AuthResponse { Success = true, Message = "횇혷ifre s횆짹f횆짹rlama linki email adresinize g횄쨋nderildi." };
    }

    public async Task<AuthResponse> ResetPasswordAsync(ResetPasswordRequest request)
    {
        var user = await _userManager.FindByEmailAsync(request.Email);
        if (user == null)
            return new AuthResponse { Success = false, Message = "Kullan횆짹c횆짹 bulunamad횆짹." };

        var isValid = await _otpService.ValidateCodeAsync(user.Id, request.Token, "password_reset");
        if (!isValid)
            return new AuthResponse { Success = false, Message = "Ge횄짠ersiz veya s횄쩌resi dolmu횇혺 token." };

        var removeResult = await _userManager.RemovePasswordAsync(user);
        if (!removeResult.Succeeded)
            return new AuthResponse { Success = false, Message = "횇혷ifre s횆짹f횆짹rlanamad횆짹." };

        var addResult = await _userManager.AddPasswordAsync(user, request.NewPassword);
        if (!addResult.Succeeded)
            return new AuthResponse
            {
                Success = false,
                Message = string.Join(", ", addResult.Errors.Select(e => e.Description))
            };

        return new AuthResponse { Success = true, Message = "횇혷ifreniz ba횇혺ar횆짹yla s횆짹f횆짹rland횆짹." };
    }

    public async Task<AuthResponse> ResendVerificationAsync(ResendVerificationRequest request)
    {
        var user = await _userManager.FindByEmailAsync(request.Email);
        if (user == null)
            return new AuthResponse { Success = true, Message = "Do횆혺rulama emaili g횄쨋nderildi." };

        if (user.EmailConfirmed)
            return new AuthResponse { Success = false, Message = "Email zaten do횆혺rulanm횆짹횇혺." };

        var token = await _otpService.GenerateAndSaveCodeAsync(user.Id, "email_verification");
        await _emailService.SendVerificationEmailAsync(user.Email!, user.Id, token);

        return new AuthResponse { Success = true, Message = "Do횆혺rulama emaili g횄쨋nderildi." };
    }
}

using BeachGo.Api.Models;
using BeachGo.Api.Services;
using FluentValidation;
using Microsoft.AspNetCore.Mvc;

namespace BeachGo.Api.Controllers;

[ApiController]
[Route("api/auth")]
public class AuthController : ControllerBase
{
    private readonly IAuthService _authService;
    private readonly IValidator<ForgotPasswordRequest> _forgotPasswordValidator;
    private readonly IValidator<ResetPasswordRequest> _resetPasswordValidator;
    private readonly IValidator<VerifyEmailRequest> _verifyEmailValidator;
    private readonly IValidator<ResendVerificationRequest> _resendVerificationValidator;

    public AuthController(
        IAuthService authService,
        IValidator<ForgotPasswordRequest> forgotPasswordValidator,
        IValidator<ResetPasswordRequest> resetPasswordValidator,
        IValidator<VerifyEmailRequest> verifyEmailValidator,
        IValidator<ResendVerificationRequest> resendVerificationValidator)
    {
        _authService = authService;
        _forgotPasswordValidator = forgotPasswordValidator;
        _resetPasswordValidator = resetPasswordValidator;
        _verifyEmailValidator = verifyEmailValidator;
        _resendVerificationValidator = resendVerificationValidator;
    }

    [HttpPost("register")]
    public async Task<IActionResult> Register([FromBody] RegisterRequest request)
    {
        var result = await _authService.RegisterAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);
    }

    [HttpPost("login")]
    public async Task<IActionResult> Login([FromBody] LoginRequest request)
    {
        var result = await _authService.LoginAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);

    {

        if (!validation.IsValid)
            return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

        var result = await _authService.VerifyEmailAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);
    }

    [HttpPost("forgot-password")]

    {
        var validation = await _forgotPasswordValidator.ValidateAsync(request);
        if (!validation.IsValid)
            return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

        var result = await _authService.ForgotPasswordAsync(request);
        return Ok(result); // Her zaman 200 d횄쨋n (g횄쩌venlik)
    }

    [HttpPost("reset-password")]
    public async Task<IActionResult> ResetPassword([FromBody] ResetPasswordRequest request)
    {
        var validation = await _resetPasswordValidator.ValidateAsync(request);
        if (!validation.IsValid)
            return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

        var result = await _authService.ResetPasswordAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);

    public async Task<IActionResult> ResendVerification([FromBody] ResendVerificationRequest request)
    {
        var validation = await _resendVerificationValidator.ValidateAsync(request);
        if (!validation.IsValid)
            return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

        var result = await _authService.ResendVerificationAsync(request);
        return result.Success ? Ok(result) : BadRequest(result);
    }
