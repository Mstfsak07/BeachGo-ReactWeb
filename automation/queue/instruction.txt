using Microsoft.EntityFrameworkCore;
using [PROJE_ADI].Data;
using [PROJE_ADI].Models;

namespace [PROJE_ADI].Services
{
    public class OtpService : IOtpService
    {
        private readonly ApplicationDbContext _context;
        private readonly ILogger<OtpService> _logger;
        private const int CODE_EXPIRY_MINUTES = 15;

        public OtpService(ApplicationDbContext context, ILogger<OtpService> logger)
        {
            _context = context;
            _logger = logger;
        }

        public async Task<string> GenerateAndSaveCodeAsync(string email, string type)
        {
            // Eski kodlarÄ± geÃ§ersiz kÄ±l
            var oldCodes = await _context.VerificationCodes
                .Where(v => v.Email == email && v.Type == type && !v.IsUsed)
                .ToListAsync();
            
            foreach (var old in oldCodes)
                old.IsUsed = true;

            // Yeni 6 haneli kod Ã¼ret
            var code = new Random().Next(100000, 999999).ToString();

            var verificationCode = new VerificationCode
            {
                Email = email,
                Code = code,
                Type = type,
                ExpiresAt = DateTime.UtcNow.AddMinutes(CODE_EXPIRY_MINUTES),
                IsUsed = false,
                CreatedAt = DateTime.UtcNow
            };

            _context.VerificationCodes.Add(verificationCode);
            await _context.SaveChangesAsync();

            _logger.LogInformation("OTP generated for {Email}, type={Type}", email, type);
            return code;
        }

        public async Task<bool> ValidateCodeAsync(string email, string code, string type)
        {
            var record = await _context.VerificationCodes
                .Where(v => v.Email == email && v.Code == code && v.Type == type && !v.IsUsed)
                .OrderByDescending(v => v.CreatedAt)
                .FirstOrDefaultAsync();

            if (record == null)
            {
                _logger.LogWarning("OTP not found for {Email}, type={Type}", email, type);
                return false;
            }

            if (record.ExpiresAt < DateTime.UtcNow)
            {
                _logger.LogWarning("OTP expired for {Email}", email);
                record.IsUsed = true;
                await _context.SaveChangesAsync();
                return false;
            }

            return true;
        }

        public async Task InvalidateCodeAsync(string email, string type)
        {
            var records = await _context.VerificationCodes
                .Where(v => v.Email == email && v.Type == type && !v.IsUsed)
                .ToListAsync();

            foreach (var r in records)
                r.IsUsed = true;

            await _context.SaveChangesAsync();
        }
    }
}

    var validator = new ForgotPasswordRequestValidator();
    var validation = await validator.ValidateAsync(request);
    if (!validation.IsValid)
        return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

    var (success, message) = await _authService.ForgotPasswordAsync(request.Email);
    return Ok(new { success, message });
}

[HttpPost("reset-password")]
[AllowAnonymous]
public async Task<IActionResult> ResetPassword([FromBody] ResetPasswordRequest request)
{
    var validator = new ResetPasswordRequestValidator();

    if (!validation.IsValid)
        return BadRequest(new { errors = validation.Errors.Select(e => e.ErrorMessage) });

    var (success, message) = await _authService.ResetPasswordAsync(request.Email, request.Token, request.NewPassword);
    if (!success)
        return BadRequest(new { success, message });

    return Ok(new { success, message });
}
